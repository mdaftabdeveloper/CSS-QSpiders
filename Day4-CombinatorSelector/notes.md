# Combinator Selector

=> Drect Child Selector (>)
-> Syntax:
    parent > child {
        property: value;
    }



=> Descendent Selector ( )
-> Syntax
    topmost innermost {
        property: value;
    }

=> Immediate Sibling Selector (+)
-> Syntax:
    first_sibling + immediate_sibling {
        property; value;
    }
